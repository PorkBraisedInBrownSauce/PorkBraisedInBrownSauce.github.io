翻墙hosts切换小程序

May 20, 2016

校园网和中科院网络一般都是支持ipv6的，此时可以使用ipv6访问youtube，google，且不要流量。但有时候出了校园网又会有些不方便，写了一个小python程序，可以实现切换和hosts更新。

# -*- coding: utf-8 -*-
"""
Created on Fri May 13 21:45:16 2016

@author: ClT
"""

import Tkinter as tk
import os
import shutil
import urllib2
path = r"C:\Windows\System32\drivers\etc"
cur = os.path.split(os.path.realpath(__file__))[0]
url = r"http://7xs6jl.com1.z0.glb.clouddn.com/hosts"
class Application(tk.Frame):

    def trepair(self):
        if os.path.isfile(os.path.join(path,"hosts")) and not os.path.isfile(os.path.join(path,"hosts.backup")):
            os.rename(os.path.join(path,"hosts"),os.path.join(path,"hosts.backup"))
            shutil.copy(os.path.join(cur,"temp"), path)
            os.rename(os.path.join(path,"temp"),os.path.join(path,"hosts"))
    def trecover(self):
        if os.path.isfile(os.path.join(path,"hosts.backup")):
            os.remove(os.path.join(path,"hosts"))
            os.rename(os.path.join(path,"hosts.backup"),os.path.join(path,"hosts"))

    def tupdate(self):
        f1 = urllib2.urlopen(url)
        data = f1.read()
        with open(os.path.join(cur,"temp"),'w') as f2:
            f2.truncate()
            f2.write(data)

    def createWidgets(self):
        self.recover = tk.Button(self)
        self.recover["text"] = "复原"
        self.recover["fg"]   = "red"
        self.recover["command"] =  self.trecover
        self.recover.grid(row=3)
        self.repair = tk.Button(self)
        self.repair["text"] = "上网"
        self.repair["fg"]   = "blue"
        self.repair["command"] = self.trepair
        self.repair.grid(row=2)     
        self.label = tk.Label(self, text = "hello")      
        self.update = tk.Button(self)
        self.update["text"] = "更新"
        self.update["command"] = self.tupdate
        self.update.grid(row=4)

    def __init__(self, master=None):
        tk.Frame.__init__(self, master)
        self.pack()
        self.createWidgets()

root = tk.Tk()
app = Application(master=root)
app.mainloop()
root.destroy()
